一、四种内存的分配释放方法

分配 释放 类别 可否重载
malloc() free() C函数 不可
new delete C++表达式 不可
::operator new() ::operator delete() C++函数 不可
allocator::allocate() allocator::deallocate() C++标准库 可自由设计并搭配任何容器

二、new&delete expression

1、new expression

对于代码：

Complex* pc = new Complex(1,2);

编译器会转换为：

Complex *pc;
try{
    //分配内存大小
    void* mem = operator new( sizeof(Complex) ); //allocate
    //强制类型转换
    pc = static_cast<Complex*>(mem);             //cast
    //调用目标类型的构造函数
    pc->Complex::Complex(1,2);                   //constract
    /*
    注意：只有编译器才可以像上面那样直接呼叫ctor(构造函数)
    欲直接调用ctor，可以运用placement new：
    new(p)Complex(1,2);
    */
}
catch( std::bad_alloc ){
    //若allocation失败就不执行constructor
}

opeator new源码（VC）:

void *operator new(size_t size,const std::nothrow_t&)_THROW0()
{
    //const std::nothrow_t& 函数保证不抛出异常
    //该结构用作函数operator new参数，表示函数应该返回一个空指针来报告分配失败。而不是抛出异常。
	//try to allocate size bytes
    void *p;
    //内部调用了malloc函数
    while((p=malloc(size))==0)
    {
        //buy more memory ot return null pointer
        _TRY_BEGIN
            if(_callnewh(size)==0) break;
        _CATCH(std::bad_alloc) return(0);
        _CATCH_END
    }
    return (p);
}

2、delete expression

对于代码：

Complex* pc = new Complex(1,2);
......
delete pc;

编译器会转为：

pc->~Complex();                 //先析构
//不同于new expression，这里可以直接调用析构函数
operatoe delete(pc);            //然后释放内存

opeator new源码（VC）:

//.../vc98/crt/src/delop.cpp
void__cdecl operator delete(void *p)_THROW0(){
    //free an aallocated object
    free(p);
    //内部调用了free函数
}

三、Array new

Complex* pca = new Complex[3];
//唤起三次ctor
//无法藉由参数给予初值
...
delete[] pca;//唤起三次dtor
//如果没带[]，则只会唤起一次dtor

如果没有对每个object都调用dtor：

  • 对于 class without ptr member 可能没有影响
  • 对于 class with pointer member（带指针的类）通常有影响—内存泄漏

所以对于上面的代码，其实调用一次dtor和调用三次dtor是相同的

而对于下面的代码：

string* psa = new string[3];
...
delete psa;//唤起一次dtor

这样只析构了一个对象，从而会导致内存泄漏

会析构第一个对象或者最后一个对象，不同的编译器会有所不同

#include <iostream>
#include <new>		//placement new
using namespace std;

namespace jj03
{

	class A
	{
	public:
		int id;

		A() : id(0)      { cout << "default ctor. this=" << this << " id=" << id << endl; }
		A(int i) : id(i) { cout << "ctor. this=" << this << " id=" << id << endl; }
		~A()             { cout << "dtor. this=" << this << " id=" << id << endl; }
	};

	void test_array_new_and_placement_new()
	{
		cout << "\ntest_placement_new().......... \n";

		size_t size = 3;

		{
			//case 1
			//模擬 memory pool 的作法, array new + placement new. 崩潰 

			A* buf = (A*)(new char[sizeof(A)*size]);
			A* tmp = buf;

			cout << "buf=" << buf << "  tmp=" << tmp << endl;

			for (int i = 0; i < size; ++i)
				new (tmp++) A(i);  			//3次 调用ctor 

			cout << "buf=" << buf << "  tmp=" << tmp << endl;

			//!	delete [] buf;    	//crash. why?
			//因為這其實是個 char array，看到 delete [] buf; 編譯器會企圖喚起多次 A::~A. 
			// 但 array memory layout 中找不到與 array 元素個數 (本例 3) 相關的信息, 
			// -- 整個格局都錯亂 (從我對 VC 的認識而言)，於是崩潰。 
			delete buf;     	//dtor just one time, ~[0]	

			cout << "\n\n";
		}

		{
			//case 2
			//回頭測試單純的 array new

			A* buf = new A[size];  //default ctor 3 次. [0]先於[1]先於[2])
			//A必須有 default ctor, 否則 [Error] no matching function for call to 'jj02::A::A()'
			A* tmp = buf;

			cout << "buf=" << buf << "  tmp=" << tmp << endl;

			for (int i = 0; i < size; ++i)
				new (tmp++) A(i);  		//3次 ctor 

			cout << "buf=" << buf << "  tmp=" << tmp << endl;

			delete[] buf;    //dtor three times (次序逆反, [2]先於[1]先於[0])	
		}

		{
			//case 3	
			//掌握崩潰原因, 再次模擬 memory pool作法, array new + placement new. 	
			//不, 不做了, 因為 memory pool 只是供應 memory, 它並不管 construction, 
			//也不管 destruction. 它只負責回收 memory. 
			//所以它是以 void* 或 char* 取得 memory, 釋放 (刪除)的也是 void* or char*.  
			//不像本例 case 1 釋放 (刪除) 的是 A*. 
			//
			//事實上 memory pool 形式如 jj04::test 
		}

	}
} //namespace

int main(void)
{
	jj03::test_array_new_and_placement_new();
	return 0;
}

四、placement new

（定位放置new）是将不同于，它是将对象构建在一个已经被分配好的内存之中，所以在使用时，需要一个已经分配好的内存。

图中的代码：

Complex* pc = new(buf)Complex(1,2);

会被编译成三步：

void* mem = operator new(sizeof(Complex),buf);  //占用分配好的内存，没有实际的意义
pc = static_cast<Complex*>(mem);				//转型
pc -> Complex:Complex(1,2);						//构造对象
//所以实际上placement new就跟调用构造函数差不多

使用语法定位生成对象时，会自动调用类A的构造函数（如上），但是由于对象的空间不会自动释放（对象实际上是借用别人的空间），所以必须显示调用类A的析构函数，所以说

没有所谓的，因为根本就没有分配内存，是借用的别人的空间。

五、重载

1、C++内存分配

默认情况下，在调用new分配内存时，系统会走下面（即②）这条路线，调用全局函数，但是如果你在类中重载了函数，则系统会优先调用你重载的函数，所以我们可以自己通过重载的方式来改变内存的分配机制，但是最后还是会调用到系统的全局函数。

另外，系统的全局函数也是可以重载的，但是比较少见，一旦重载，影响的就不仅仅局限于这个类了，所以一般不使用这种方式。

C++容器，即STL中的内存分配到头来也是调用了函数。

2、重载示例

一般不推荐重载::operator new()函数，因为它对全局有影响，如果使用不当将造成很大的问题。

与重载系统函数不同，如果在类内重载函数，可以有许多种方法，但是必须保证函数参数第一个为类型，对于函数，第一个参数为类型，第二个参数为类型（可以省略）。

但是我们一般不会重载函数，因为重载的函数无法被手动调用，如：

void operator delete(void* ptr， int num)
{
 std::cout << "operator delete" << std::endl;
 std::free(ptr);
}

delete(10) p;    // 不合法的

因为重载函数意义为与重载的函数配套，只有函数报异常了，才会调用对应的，若此时没有对应的，则无法释放内存。

我自己的实践也是如上，会报未经处理的异常，不会调用重载的函数。

六、pre-class allocator

示例1

每个对象大小为，代码中通过预先分配一大块内存（类似内存池），用单向链表串连起来，每个对象间间隔个，而右边则是分别多个对象，每个对象的内存中还要分配信息，头尾各一个各，所以间隔。

代码：

#include <cstddef>
#include <iostream>
using namespace std;

namespace jj04
{
    //ref. C++Primer 3/e, p.765
    //per-class allocator 

    class Screen {
    public:
        Screen(int x) : i(x) { };
        int get() { return i; }

        void* operator new(size_t);
        void  operator delete(void*, size_t);   //(2)
        //! void  operator delete(void*);           //(1) 二擇一. 若(1)(2)並存,會有很奇怪的報錯 (摸不著頭緒) 

    private:
        Screen* next;
        static Screen* freeStore;
        static const int screenChunk;
    private:
        int i;
    };
    Screen* Screen::freeStore = 0;
    const int Screen::screenChunk = 24;

    void* Screen::operator new(size_t size)
    {
        Screen *p;
        if (!freeStore) {
            //linked list 是空的，所以攫取一大塊 memory
            //以下呼叫的是 global operator new
            size_t chunk = screenChunk * size;
            freeStore = p =
                reinterpret_cast<Screen*>(new char[chunk]);
            //將分配得來的一大塊 memory 當做 linked list 般小塊小塊串接起來
            for (; p != &freeStore[screenChunk - 1]; ++p)
                p->next = p + 1;
            p->next = 0;
        }
        p = freeStore;
        freeStore = freeStore->next;
        return p;
    }
        //! void Screen::operator delete(void *p)       //(1)
    void Screen::operator delete(void *p, size_t)   //(2)二擇一 
    {
        //將 deleted object 收回插入 free list 前端
        (static_cast<Screen*>(p))->next = freeStore;
        freeStore = static_cast<Screen*>(p);
    }

    //-------------
    void test_per_class_allocator_1()
    {
        cout << "\ntest_per_class_allocator_1().......... \n";

        cout << sizeof(Screen) << endl;     //8 

        size_t const N = 100;
        Screen* p[N];

        for (int i = 0; i< N; ++i)
            p[i] = new Screen(i);

        //輸出前 10 個 pointers, 用以比較其間隔 
        for (int i = 0; i< 10; ++i)
            cout << p[i] << endl;

        for (int i = 0; i< N; ++i)
            delete p[i];
    }
} //namespace

int main(void)
{
    jj04::test_per_class_allocator_1();
    return 0;
}
